# 简介
本章节讲解如何将第三方组件封装为vue的形式组件。

# 1、组件封装基础
使用count-up组件演示如何封装一个第三方组件

**1、下载包**
```
cnpm install countup.js -S
```
**2、创建组件包**
```js
// components/count-to/
<template>
  <div>
    <span :id="elementId"></span>
  </div>
</template>
<script>
// 引入countup
import { CountUp } from 'countup'

// 要保证每个组件实例的ID不相同
export default {
  name: 'CountTo',
  computed: {
    // 唯一ID
    elementId () {
      return `count_up_${this._uid}`
    }
  },
  data () {
    return {
      counter: {}
    }
  },
  props: {
    // 起始值
    startVal: {
      type: Number,
      default: 0
    },
    // 终止值
    endVal: {
      type: Number,
      default: 100
    },
    // 小数点几位
    decimals: {
      type: Number,
      default: 0
    },
    // 渐变时长
    duration: {
      type: Number,
      default: 3
    },
    // 是否使用变速效果
    useEasing: {
      type: Boolean,
      default: false
    },
    // 是否使用分组
    useGrouping: {
      type: Boolean,
      default: false
    },
    // 分组符号
    separator: {
      type: String,
      default: ','
    },
    // 整数和小数的分隔符号
    decimal: {
      type: String,
      default: '.'
    },
    // 延迟动画时间
    delay: {
      type: Number,
      default: 0
    }
  },
  mounted () {
    // 创建三方组件实例
    console.log(this.endVal)
    this.$nextTick(() => {
      // this.counter = new CountUp(this.elementId, this.startVal, this.endVal, this.decimals, this.duration, {
      //   useEasing: this.useEasing,
      //   useGrouping: this.useGrouping,
      //   separator: this.separator,
      //   decimal: this.decimal
      // })

      this.counter = new CountUp(this.elementId, this.endVal, {
        startVal: this.startVal,
        decimals: this.decimals,
        duration: this.duration,
        useEasing: this.useEasing,
        useGrouping: this.useGrouping,
        separator: this.separator,
        decimal: this.decimal
      })

      console.log(this.counter)

      // 延迟效果实现
      setTimeout(() => {
        this.counter.start()
      }, this.delay)
    })
  }
}
</script>
```
**3、创建组件包的快速引入**
```js
// components/count-to/index.js
import CountTo from './count-to.vue'

export default CountTo
```

**4、使用组件**
```js
// views/count-to.vue
<template>
  <div>
    <count-to :end-val='100'></count-to>
  </div>
</template>

<script>
// 由于编写了index.js，此处可以直接引入
import CountTo from '@/components/count-to'

export default {
  name: 'count_to',
  components: {
    CountTo
  }
}
</script>
```
